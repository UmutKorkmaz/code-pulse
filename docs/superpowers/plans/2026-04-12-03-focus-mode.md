# Focus Mode / Pomodoro Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a built-in Pomodoro timer that integrates with tracking: during focus, the status bar indicates "In Focus", breaks auto-pause tracking, and events fire for the Notifications system (next plan) to consume.

**Architecture:** A new `FocusManager` class owns a state machine: `idle → focusing → break → idle`. It exposes an `EventBus` (simple typed emitter) so other modules can react without coupling. `StatusBarManager` subscribes to render the focus phase. Tracking pauses/resumes via `timeTracker.stop()`/`start()` during breaks.

**Tech Stack:** TypeScript. No new dependencies. Uses native `setTimeout` for timers.

```typescript
type Handler<T> = (payload: T) => void;

export class EventBus<Events extends Record<string, unknown>> {
    private handlers = new Map<keyof Events, Set<Handler<never>>>();

    on<K extends keyof Events>(event: K, fn: Handler<Events[K]>): void {
        let set = this.handlers.get(event);
        if (!set) { set = new Set(); this.handlers.set(event, set); }
        set.add(fn as Handler<never>);
    }

    off<K extends keyof Events>(event: K, fn: Handler<Events[K]>): void {
        this.handlers.get(event)?.delete(fn as Handler<never>);
    }

    emit<K extends keyof Events>(event: K, payload: Events[K]): void {
        const set = this.handlers.get(event);
        if (!set) return;
        for (const fn of set) {
            try { (fn as Handler<Events[K]>)(payload); }
            catch (e) { console.error(`EventBus handler error for ${String(event)}:`, e); }
        }
    }
}

export interface CodePulseEvents {
    'session-started': { sessionId: string };
    'session-ended': { sessionId: string; durationMs: number };
    'focus-started': { focusMs: number };
    'focus-break': { breakMs: number };
    'focus-completed': { pomodoroCount: number };
    'long-session': { durationMs: number };
    'goal-met': { hours: number };
}
```

### Task 4: Implement FocusManager

**Files:** Create `src/focus/FocusManager.ts`

- [ ] **Step 1: Implement the class**

```typescript
import { ConfigManager } from '../utils/ConfigManager';
import { EventBus, CodePulseEvents } from '../events/EventBus';

export type FocusState = 'idle' | 'focusing' | 'break';

export class FocusManager {
    private state: FocusState = 'idle';
    private timer?: NodeJS.Timeout;
    private pomodoroCount = 0;
    private currentPhaseEndsAt?: number;

    constructor(
        private configManager: ConfigManager,
        private bus: EventBus<CodePulseEvents>
    ) {}

    getState(): FocusState { return this.state; }
    getPhaseEndsAt(): number | undefined { return this.currentPhaseEndsAt; }
    getPomodoroCount(): number { return this.pomodoroCount; }

    start(): void {
        const focusMin = this.configManager.get<number>('focus.pomodoroMinutes', 25);
        const breakMin = this.configManager.get<number>('focus.breakMinutes', 5);
        this.startWithDurations(focusMin, breakMin);
    }

    /** Test hook — takes durations in minutes (can be fractional). */
    startWithDurations(focusMin: number, breakMin: number): void {
        if (this.state !== 'idle') return;
        this.state = 'focusing';
        const focusMs = Math.round(focusMin * 60 * 1000);
        this.currentPhaseEndsAt = Date.now() + focusMs;
        this.bus.emit('focus-started', { focusMs });
        this.timer = setTimeout(() => this.enterBreak(breakMin), focusMs);
    }

    stop(): void {
        if (this.timer) { clearTimeout(this.timer); this.timer = undefined; }
        this.state = 'idle';
        this.currentPhaseEndsAt = undefined;
    }

    skipBreak(): void {
        if (this.state !== 'break') return;
        if (this.timer) { clearTimeout(this.timer); this.timer = undefined; }
        this.state = 'idle';
        this.currentPhaseEndsAt = undefined;
        this.start();
    }

    private enterBreak(breakMin: number): void {
        this.pomodoroCount += 1;
        this.state = 'break';
        const longBreakMin = this.configManager.get<number>('focus.longBreakMinutes', 15);
        const effectiveMin = this.pomodoroCount % 4 === 0 ? longBreakMin : breakMin;
        const breakMs = Math.round(effectiveMin * 60 * 1000);
        this.currentPhaseEndsAt = Date.now() + breakMs;
        this.bus.emit('focus-break', { breakMs });
        this.bus.emit('focus-completed', { pomodoroCount: this.pomodoroCount });
        this.timer = setTimeout(() => {
            this.state = 'idle';
            this.currentPhaseEndsAt = undefined;
        }, breakMs);
    }
}
```

eventBus.on('focus-break', () => {
    if (configManager.get<boolean>('focus.autoPauseTracking', true)) {
        timeTracker.stop().catch(err => logger.warn('Failed to pause tracking for break', err instanceof Error ? err : new Error(String(err))));
    }
});

eventBus.on('focus-started', () => {
    if (configManager.get<boolean>('focus.autoPauseTracking', true) && !timeTracker.isEnabled()) {
        // do nothing — user disabled tracking
        return;
    }
    timeTracker.start().catch(err => logger.warn('Failed to resume tracking for focus', err instanceof Error ? err : new Error(String(err))));
});
