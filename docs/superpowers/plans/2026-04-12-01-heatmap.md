# Contribution Heatmap Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Render a GitHub-style 365-day heatmap on the dashboard, with each cell showing coding intensity for one day. Clicking a cell filters the Sessions Table to that day.

**Architecture:** Query the existing `daily_rollups` table (pre-aggregated by date), bucket each day's total time into 5 intensity levels, build an SVG grid by DOM construction (no innerHTML strings with user data). No new DB schema, no migrations. All theme-driven colors.

**Tech Stack:** TypeScript backend (one new method on `AnalyticsEngine`) + vanilla JS/SVG in `dashboard.js` using `document.createElementNS` + CSS in `dashboard.css`.

```typescript
/**
 * Bucket a daily duration (ms) into a 0-4 intensity level for heatmap cells.
 */
export function intensityBucket(totalMs: number): 0 | 1 | 2 | 3 | 4 {
    const hours = totalMs / (60 * 60 * 1000);
    if (hours <= 0) return 0;
    if (hours <= 1) return 1;
    if (hours <= 3) return 2;
    if (hours <= 6) return 3;
    return 4;
}
```

```typescript
public async getHeatmapData(startDate: Date, endDate: Date): Promise<HeatmapCell[]> {
    const dates = eachLocalDate(startDate, endDate);
    const sessions = await this.databaseManager.getSessionsByDateRange(startDate, endDate);
    const byDate = new Map<string, { totalMs: number; count: number }>();

    for (const s of sessions) {
        const key = formatLocalDate(s.startTime);
        const cur = byDate.get(key) || { totalMs: 0, count: 0 };
        cur.totalMs += s.duration;
        cur.count += 1;
        byDate.set(key, cur);
    }

    return dates.map(date => {
        const agg = byDate.get(date) || { totalMs: 0, count: 0 };
        return {
            date,
            totalMs: agg.totalMs,
            intensity: intensityBucket(agg.totalMs),
            sessionCount: agg.count
        };
    });
}
```

```typescript
private async getHeatmapData(days: number, webview?: vscode.Webview) {
    try {
        const endDate = new Date();
        const startDate = new Date();
        startDate.setDate(endDate.getDate() - days);
        const { AnalyticsEngine } = await import('../analytics/AnalyticsEngine');
        const engine = new AnalyticsEngine(this.databaseManager, this.configManager);
        const cells = await engine.getHeatmapData(startDate, endDate);
        const target = webview ?? this._view?.webview;
        target?.postMessage({ command: 'updateHeatmapData', data: { cells, days } });
    } catch (error) {
        console.error('Failed to get heatmap data:', error);
    }
}
```

```javascript
function renderHeatmap(payload) {
    const svg = document.getElementById('heatmap');
    const summary = document.getElementById('heatmapSummary');
    if (!svg || !payload) return;

    while (svg.firstChild) svg.removeChild(svg.firstChild);
    const cells = payload.cells || [];
    if (cells.length === 0) {
        if (summary) summary.textContent = 'No data';
        return;
    }

    const NS = 'http://www.w3.org/2000/svg';
    const cellSize = 12;
    const cellGap = 3;
    const pitch = cellSize + cellGap;
    const leftPad = 26;
    const topPad = 18;

    const first = new Date(cells[0].date);
    const paddingCells = first.getDay();
    const weeks = Math.ceil((cells.length + paddingCells) / 7);
    const width = leftPad + weeks * pitch;
    const height = topPad + 7 * pitch;
    svg.setAttribute('viewBox', `0 0 ${width} ${height}`);
    svg.setAttribute('width', String(width));
    svg.setAttribute('height', String(height));

    // Weekday labels (Mon / Wed / Fri only, for legibility)
    [[1, 'Mon'], [3, 'Wed'], [5, 'Fri']].forEach(([row, label]) => {
        const t = document.createElementNS(NS, 'text');
        t.setAttribute('class', 'heatmap-day-label');
        t.setAttribute('x', '0');
        t.setAttribute('y', String(topPad + row * pitch + cellSize - 2));
        t.textContent = label;
        svg.appendChild(t);
    });

    // Month labels
    let lastMonth = -1;
    cells.forEach((cell, i) => {
        const d = new Date(cell.date);
        const pos = i + paddingCells;
        const col = Math.floor(pos / 7);
        if (d.getMonth() !== lastMonth && d.getDate() <= 7) {
            const t = document.createElementNS(NS, 'text');
            t.setAttribute('class', 'heatmap-month-label');
            t.setAttribute('x', String(leftPad + col * pitch));
            t.setAttribute('y', String(topPad - 4));
            t.textContent = d.toLocaleDateString(undefined, { month: 'short' });
            svg.appendChild(t);
            lastMonth = d.getMonth();
        }
    });

    // Cells (via DOM, no innerHTML)
    let total = 0, activeDays = 0;
    cells.forEach((cell, i) => {
        const pos = i + paddingCells;
        const col = Math.floor(pos / 7);
        const row = pos % 7;
        const rect = document.createElementNS(NS, 'rect');
        rect.setAttribute('class', 'heatmap-cell');
        rect.setAttribute('data-date', cell.date);
        rect.setAttribute('data-intensity', String(cell.intensity));
        rect.setAttribute('x', String(leftPad + col * pitch));
        rect.setAttribute('y', String(topPad + row * pitch));
        rect.setAttribute('width', String(cellSize));
        rect.setAttribute('height', String(cellSize));
        rect.setAttribute('rx', '2');
        rect.setAttribute('ry', '2');

        const title = document.createElementNS(NS, 'title');
        const hours = (cell.totalMs / (60 * 60 * 1000)).toFixed(1);
        title.textContent = `${cell.date}: ${hours}h over ${cell.sessionCount} session${cell.sessionCount === 1 ? '' : 's'}`;
        rect.appendChild(title);

        rect.addEventListener('click', () => {
            if (!sessionsState) return;
            sessionsState.filters.search = cell.date;
            const input = document.getElementById('sessionSearch');
            if (input) input.value = cell.date;
            applySessionsFilters();
            document.querySelector('.sessions-card')?.scrollIntoView({ behavior: 'smooth', block: 'start' });
        });

        svg.appendChild(rect);

        total += cell.totalMs;
        if (cell.totalMs > 0) activeDays++;
    });

    if (summary) {
        const hoursTotal = (total / (60 * 60 * 1000)).toFixed(1);
        summary.textContent = `${activeDays} active days · ${hoursTotal}h total`;
    }
}
```
